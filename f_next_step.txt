  WHERE user_id=p_user_id AND id_empresa=p_id_empresa

  END IF;

    );
  END IF;

      'acao','pedir_qtd',
      'mensagem','Quantas unidades vocÃª quer?'
    );
  END IF;

  END IF;

  END IF;
