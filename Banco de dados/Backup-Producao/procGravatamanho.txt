 CREATE OR REPLACE FUNCTION GravaTamanho(
  p_user_id     integer,
  p_id_empresa  integer,
  p_tamanho        text
)
RETURNS text
LANGUAGE plpgsql
AS $$
BEGIN
  UPDATE turno_opcoes
     SET tamanho = p_tamanho        
   WHERE user_id    = p_user_id
     AND id_empresa = p_id_empresa;

  RETURN 'ok';
END;
$$;
