CREATE OR REPLACE FUNCTION limpar_opcoes_turno(
  p_user_id      integer,
  p_pedido_id    bigint,
  p_id_empresa   int,
  p_contexto     text
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
  DELETE FROM turno_opcoes
   WHERE user_id = p_user_id
     AND pedido_id = p_pedido_id
     AND id_empresa = p_id_empresa
     AND contexto = p_contexto;
END;
$$;
