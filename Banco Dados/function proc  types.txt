  CREATE OR REPLACE FUNCTION public.finalizar_pedido(p_user_id integer, p_pedido_id integer)
 RETURNS text
 LANGUAGE plpgsql
AS $function$
DECLARE
  v_pedido_id bigint;  
  v_total     numeric;
  v_pagto     text;
  v_itens     text;
  v_resumo    text;
  v_comentario text;
  v_nome      text;
  v_endereco  text;
  v_bairro    text;
BEGIN
  -- Atualiza valor no pedido_temp
  UPDATE public.pedido_temp p
     SET valor = COALESCE((
       SELECT SUM(i.valor)
         FROM public.item_pedido_temp i
        WHERE i.user_id   = p_user_id
          AND i.pedido_id = p_pedido_id
     ),0) + 3
   WHERE p.user_id   = p_user_id
     AND p.pedido_id = p_pedido_id;

  -- Busca totais, itens e comentário
  SELECT
    SUM(i.valor) + 3 AS total,
    MAX(p.tipo_cobranca) AS tipo_cobranca,
    STRING_AGG(
      E'\n       '   
      TRIM(BOTH ' ' FROM
        CASE WHEN i.quantidade > 1 THEN i.quantidade::text    'x ' ELSE '' END   
        i.nome   
        CASE WHEN COALESCE(i.tamanho,'') <> '' THEN ' ('    i.tamanho    ')' ELSE '' END   
        CASE WHEN COALESCE(i.categoria,'') <> '' THEN ' - '    i.categoria ELSE '' END   
        ' - R$ '    TO_CHAR(i.valor, 'FM999G999G990D00')
      ),
      '' ORDER BY i.categoria, i.nome
    ) AS itens,
    COALESCE(NULLIF(BTRIM(c.comentario),''),'—') AS comentario
  INTO v_total, v_pagto, v_itens, v_comentario
  FROM public.item_pedido_temp i
  LEFT JOIN public.pedido_temp p
         ON p.user_id   = i.user_id
        AND p.pedido_id = i.pedido_id
  LEFT JOIN public.comentario_pedido_temp c
         ON c.user_id   = i.user_id
        AND c.pedido_id = i.pedido_id
 WHERE i.user_id   = p_user_id
   AND i.pedido_id = p_pedido_id
 GROUP BY c.comentario;

  IF v_total IS NULL THEN
    RAISE EXCEPTION 'Pedido temporário % do usuário % não encontrado ou sem itens.', p_pedido_id, p_user_id;
  END IF;

  -- Busca dados do cliente
  SELECT u.nome, u.endereço, u.bairro
    INTO v_nome, v_endereco, v_bairro
    FROM usuario_lacantina u
   WHERE u.user_id = p_user_id
   LIMIT 1;

  -- Insere em pedidos
  INSERT INTO public.pedidos (user_id, status, valor, resumo, tipo_cobranca)
  SELECT user_id, 'recebido', COALESCE(valor, v_total), NULL, tipo_cobranca
    FROM public.pedido_temp
   WHERE user_id   = p_user_id
     AND pedido_id = p_pedido_id
  RETURNING pedido_id INTO v_pedido_id;

  -- Monta resumo já com o NOVO pedido_id
  v_resumo :=
      '📦 Pedido nº '    v_pedido_id    E'\n'   
      '📋 Resumo: '      COALESCE(v_itens,'')    E'\n'   
      '💳 Forma de pagamento: '    COALESCE(v_pagto,'não informado')    E'\n'   
      '📦 Entrega: R$3,00'    E'\n'   
      '💰 Total: R$ '    TO_CHAR(v_total,'FM999G999G990D00')    E'\n'   
      '💬 Comentário: '    v_comentario    E'\n'   
      '👤 Cliente: '      COALESCE(v_nome,'(não cadastrado)')    E'\n'   
      '🏠 Endereço: '     COALESCE(v_endereco,'(sem endereço)')    ' – '    COALESCE(v_bairro,'(sem bairro)');

  -- Insere itens finais
  INSERT INTO public.item_pedido (
    numero, user_id, quantidade, tamanho, fracionada, tipo,
    categoria, numero_fracao, nome, volume, status, valor,
    valor_fracao, pedido_id, valor_unitario
  )
  SELECT numero, user_id, quantidade, tamanho, fracionada, tipo,
         categoria, numero_fracao, nome, volume, status, valor,
         valor_fracao, v_pedido_id, valor_unitario
    FROM public.item_pedido_temp
   WHERE user_id   = p_user_id
     AND pedido_id = p_pedido_id;

  -- Move comentários
  INSERT INTO public.comentario_pedido (user_id,pedido_id,comentario)
  SELECT user_id, v_pedido_id, comentario
    FROM public.comentario_pedido_temp
   WHERE user_id   = p_user_id
     AND pedido_id = p_pedido_id;

  -- Atualiza resumo
  UPDATE public.pedidos SET resumo = v_resumo WHERE pedido_id = v_pedido_id;

  -- Limpa temporários
  DELETE FROM public.comentario_pedido_temp WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
  DELETE FROM public.item_pedido_temp      WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
  DELETE FROM public.pedido_temp           WHERE user_id=p_user_id AND pedido_id=p_pedido_id;

  RETURN v_resumo;
END;

  CREATE OR REPLACE PROCEDURE public.reordenaritensconcluidos(IN p_user_id bigint, IN p_pedido_id bigint)
 LANGUAGE plpgsql
AS $procedure$
BEGIN
  WITH ordenado AS (
    SELECT 
      item_id,
      user_id,
      pedido_id,
      ROW_NUMBER() OVER (
        PARTITION BY user_id, pedido_id 
        ORDER BY item_id
      ) AS nova_ordem
    FROM item_pedido_temp
    WHERE user_id   = p_user_id
      AND pedido_id = p_pedido_id
      AND status    = 'concluido'
  )
  UPDATE item_pedido_temp i
     SET ordem_item = o.nova_ordem
  FROM ordenado o
  WHERE i.item_id   = o.item_id
    AND i.user_id   = o.user_id
    AND i.pedido_id = o.pedido_id;
END;

  CREATE OR REPLACE PROCEDURE public.valida_nome_item_pedido(IN p_user integer, IN p_pedido integer)
 LANGUAGE plpgsql
AS $procedure$
BEGIN
  -- Pizza pelo campo nome
  UPDATE item_pedido_temp AS i
  SET status = CASE
                 WHEN sub.ct = 0 THEN 'rejeitado'
                 ELSE 'opcoes'
               END
  FROM (
    SELECT i2.item_id, COUNT(c.numero) AS ct
    FROM item_pedido_temp AS i2
    LEFT JOIN cardapio AS c
      ON c.categoria = 'pizza'
     AND c.nome ILIKE '%'    TRIM(i2.nome)    '%'
    WHERE i2.status = 'inicio'
      AND i2.user_id = p_user
      AND i2.pedido_id = p_pedido
      AND i2.numero = 0
      AND i2.categoria = 'pizza'
    GROUP BY i2.item_id
    HAVING COUNT(c.numero) = 0 OR COUNT(c.numero) > 1
  ) sub
  WHERE i.item_id = sub.item_id
    AND i.status = 'inicio'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido
    AND i.numero = 0
    AND i.categoria = 'pizza';

  -- Item
  UPDATE item_pedido_temp AS i
  SET status = CASE
                 WHEN sub.ct = 0 THEN 'rejeitado'
                 ELSE 'opcoes'
               END
  FROM (
    SELECT i2.item_id, COUNT(c.numero) AS ct
    FROM item_pedido_temp AS i2
    LEFT JOIN cardapio AS c
      ON c.categoria = 'item'
     AND c.nome ILIKE '%'    TRIM(i2.nome)    '%'
    WHERE i2.status = 'inicio'
      AND i2.user_id = p_user
      AND i2.pedido_id = p_pedido
      AND i2.numero = 0
      AND i2.categoria = 'item'
    GROUP BY i2.item_id
    HAVING COUNT(c.numero) = 0 OR COUNT(c.numero) > 1
  ) sub
  WHERE i.item_id = sub.item_id
    AND i.status = 'inicio'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido
    AND i.numero = 0
    AND i.categoria = 'item';

  -- Pizza pelo campo nome_fracao
  UPDATE item_pedido_temp AS i
  SET status = CASE
                 WHEN sub.ct = 0 THEN 'rejeitado'
                 ELSE 'opcoes'
               END
  FROM (
    SELECT i2.item_id, COUNT(c.numero) AS ct
    FROM item_pedido_temp AS i2
    LEFT JOIN cardapio AS c
      ON c.categoria = 'pizza'
     AND c.nome ILIKE '%'    TRIM(i2.nome_fracao)    '%'
    WHERE i2.status = 'inicio'
      AND i2.user_id = p_user
      AND i2.pedido_id = p_pedido
      AND i2.numero_fracao = 0
      AND i2.categoria = 'pizza'
      AND i2.fracionada = true
    GROUP BY i2.item_id
    HAVING COUNT(c.numero) = 0 OR COUNT(c.numero) > 1
  ) sub
  WHERE i.item_id = sub.item_id
    AND i.status = 'inicio'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido
    AND i.numero_fracao = 0
    AND i.categoria = 'pizza'
    AND i.fracionada = true;

  -- Bebida
  UPDATE item_pedido_temp AS i
  SET status = CASE
                 WHEN sub.ct = 0 THEN 'rejeitado'
                 ELSE 'opcoes'
               END
  FROM (
    SELECT i2.item_id, COUNT(c.numero) AS ct
    FROM item_pedido_temp AS i2
    LEFT JOIN cardapio AS c
      ON c.categoria = 'bebida'
     AND (c.palavras_chave ILIKE '%'    TRIM(i2.nome)    '%')
     AND LOWER(TRIM(c.volume)) = LOWER(TRIM(i2.volume))
    WHERE i2.status = 'inicio'
      AND i2.user_id = p_user
      AND i2.pedido_id = p_pedido
      AND i2.numero = 0
      AND i2.categoria = 'bebida'
    GROUP BY i2.item_id
    HAVING COUNT(c.numero) = 0 OR COUNT(c.numero) > 1
  ) sub
  WHERE i.item_id = sub.item_id
    AND i.status = 'inicio'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido
    AND i.numero = 0
    AND i.categoria = 'bebida';

  -- Borda
  UPDATE item_pedido_temp AS i
  SET status = CASE
                 WHEN sub.ct = 0 THEN 'rejeitado'
                 ELSE 'opcoes'
               END
  FROM (
    SELECT i2.item_id, COUNT(c.numero) AS ct
    FROM item_pedido_temp AS i2
    LEFT JOIN cardapio AS c
      ON c.categoria  in (  'borda','esfirra')
     AND (c.palavras_chave ILIKE '%'    TRIM(i2.nome)    '%')
    WHERE i2.status = 'inicio'
      AND i2.user_id = p_user
      AND i2.pedido_id = p_pedido
      AND i2.numero = 0
      AND i2.categoria in (  'borda','esfirra')
    GROUP BY i2.item_id
    HAVING COUNT(c.numero) = 0 OR COUNT(c.numero) > 1
  ) sub
  WHERE i.item_id = sub.item_id
    AND i.status = 'inicio'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido
    AND i.numero = 0
    AND i.categoria  in (  'borda','esfirra');

END;

  CREATE OR REPLACE FUNCTION public.valida_pizzas_resumo(p_user_id integer, p_pedido_id integer)
 RETURNS TABLE(user_id integer, pedido_id integer, descricao text)
 LANGUAGE plpgsql
AS $function$
BEGIN
  RETURN QUERY
  WITH candidatos AS (
      -- Itens "normais"
      SELECT
          'normal'::text       AS tipo,
          i.nome               AS item_nome,
          c.numero::text       AS numero,
          c.nome               AS nome_cardapio
      FROM item_pedido_temp i
      LEFT JOIN cardapio c
             ON c.categoria = 'pizza'
            AND c.nome ILIKE '%'    i.nome    '%'
      WHERE i.status    = 'inicio'
        AND i.user_id   = p_user_id
        AND i.pedido_id = p_pedido_id
        AND i.numero    = 0
        AND i.fracionado = false
        AND COALESCE(i.nome,'') <> ''

      UNION ALL

      -- Itens "fracionados"
      SELECT
          'fracionado'::text   AS tipo,
          i.nome_fracao        AS item_nome,
          c.numero::text       AS numero,
          c.nome               AS nome_cardapio
      FROM item_pedido_temp i
      LEFT JOIN cardapio c
             ON c.categoria = 'pizza'
            AND c.nome ILIKE '%'    i.nome_fracao    '%'
      WHERE i.status    = 'inicio'
        AND i.user_id   = p_user_id
        AND i.pedido_id = p_pedido_id
        AND i.numero    = 0
        AND i.fracionado = true
        AND COALESCE(i.nome_fracao,'') <> ''
  ),
  agrupado AS (
      SELECT
          tipo,
          item_nome,
          COUNT(DISTINCT numero)                                           AS qtd,       -- 0, 1, >1
          STRING_AGG(DISTINCT numero, ', ' ORDER BY numero)                AS numeros,
          STRING_AGG(DISTINCT nome_cardapio, '   ' ORDER BY nome_cardapio) AS nomes
      FROM candidatos
      GROUP BY tipo, item_nome
  ),
  invalidos AS (
      -- Problema se não houver exatamente 1 match
      SELECT tipo, item_nome, qtd, numeros, nomes
      FROM agrupado
      WHERE qtd <> 1
  ),
  resumo AS (
      SELECT
        CASE
          WHEN COUNT(*) > 0 THEN
            'NEGADO: '   
            STRING_AGG(
              FORMAT('(%s) %s → matches=%s [%s]',
                     tipo,
                     COALESCE(item_nome, '-'),
                     qtd,
                     COALESCE(numeros, 'nenhum')),
              ' ; '
            )
          ELSE
            'OK: todos os itens válidos'
        END AS msg
      FROM invalidos
  )
  SELECT
      p_user_id    AS user_id,
      p_pedido_id  AS pedido_id,
      (SELECT msg FROM resumo) AS descricao;
END;

  CREATE OR REPLACE PROCEDURE public.reordenaritensconcluidos(IN p_user_id integer, IN p_pedido_id integer)
 LANGUAGE plpgsql
AS $procedure$
BEGIN
  WITH ordenado AS (
    SELECT 
      item_id,
      user_id,
      pedido_id,
      ROW_NUMBER() OVER (
        PARTITION BY user_id, pedido_id 
        ORDER BY item_id
      ) AS nova_ordem
    FROM item_pedido_temp
    WHERE user_id   = p_user_id
      AND pedido_id = p_pedido_id
      AND status    = 'concluido'
  )
  UPDATE item_pedido_temp i
     SET ordem_item = o.nova_ordem
  FROM ordenado o
  WHERE i.item_id   = o.item_id
    AND i.user_id   = o.user_id
    AND i.pedido_id = o.pedido_id;
END;

  CREATE OR REPLACE PROCEDURE public.popula_item_opcoes(IN p_user_id integer, IN p_pedido_id integer)
 LANGUAGE plpgsql
AS $procedure$
BEGIN
  -- Limpa opções antigas

DELETE FROM item_opcoes o
USING item_pedido_temp i
WHERE o.item_id = i.item_id
  AND i.user_id = p_user_id
  AND i.pedido_id = p_pedido_id;

  -- Pizzas inteiras
 INSERT INTO item_opcoes (item_id, numero, nome, descricao, preco, opcao, parte)
WITH cand AS (
  SELECT DISTINCT i2.item_id, c.numero, c.nome, c.descricao, c.preco_grande AS preco,  1 as parte 
  FROM item_pedido_temp i2
  JOIN cardapio c
    ON c.categoria = 'pizza'
   AND c.nome ILIKE '%'    TRIM(i2.nome)    '%'
  WHERE i2.status = 'opcoes'
    AND i2.user_id = p_user_id
    AND i2.pedido_id = p_pedido_id
    AND i2.numero = 0
    AND i2.categoria = 'pizza'
),
novos AS (
  SELECT c.*
  FROM cand c
  WHERE NOT EXISTS (
    SELECT 1 FROM item_opcoes io
    WHERE io.item_id = c.item_id AND io.numero = c.numero
  )
),
base AS (
  SELECT item_id, COALESCE(MAX(opcao),0) AS max_opcao
  FROM item_opcoes
  WHERE item_id IN (SELECT item_id FROM novos)
  GROUP BY item_id
)
SELECT n.item_id, n.numero, n.nome, n.descricao, n.preco,
       COALESCE(b.max_opcao,0) + ROW_NUMBER() OVER (PARTITION BY n.item_id ORDER BY n.numero) AS opcao, n.parte
FROM novos n
LEFT JOIN base b USING (item_id);

  -- Pizzas fracionadas (parte com nome_fracao)
 
INSERT INTO item_opcoes (item_id, numero, nome, descricao, preco, opcao, parte )
WITH cand AS (
  SELECT DISTINCT i2.item_id, c.numero, c.nome, c.descricao, c.preco_grande AS preco, 2 as parte 
  FROM item_pedido_temp i2
  JOIN cardapio c
    ON c.categoria = 'pizza'
   AND c.nome ILIKE '%'    TRIM(i2.nome_fracao)    '%'
  WHERE i2.status = 'opcoes'
    AND i2.user_id = p_user_id
    AND i2.pedido_id = p_pedido_id
    AND i2.numero_fracao = 0
    AND i2.categoria = 'pizza'
    AND i2.fracionada = true
),
novos AS (
  SELECT c.*
  FROM cand c
  WHERE NOT EXISTS (
    SELECT 1 FROM item_opcoes io
    WHERE io.item_id = c.item_id AND io.numero = c.numero
  )
),
base AS (
  SELECT item_id, COALESCE(MAX(opcao),0) AS max_opcao
  FROM item_opcoes
  WHERE item_id IN (SELECT item_id FROM novos)
  GROUP BY item_id
)
SELECT n.item_id, n.numero, n.nome, n.descricao, n.preco,
       COALESCE(b.max_opcao,0) + ROW_NUMBER() OVER (PARTITION BY n.item_id ORDER BY n.numero) AS opcao, n.parte
FROM novos n
LEFT JOIN base b USING (item_id);



  -- Bebidas e Bordas
 

INSERT INTO item_opcoes (item_id, numero, nome, descricao, preco, opcao, parte )
WITH cand AS (
  SELECT DISTINCT i2.item_id, c.numero, c.nome, c.descricao, c.preco_grande AS preco, 1 as parte 
  FROM item_pedido_temp i2
  JOIN cardapio c
    ON c.categoria IN ('bebida','borda')
   AND c.palavras_chave ILIKE '%'    TRIM(i2.nome)    '%'
  WHERE i2.status = 'opcoes'
    AND i2.user_id = p_user_id
    AND i2.pedido_id = p_pedido_id
    AND i2.numero = 0
    AND i2.categoria IN ('bebida','borda')
    AND i2.fracionada = false
),
novos AS (
  SELECT c.*
  FROM cand c
  WHERE NOT EXISTS (
    SELECT 1 FROM item_opcoes io
    WHERE io.item_id = c.item_id AND io.numero = c.numero
  )
),
base AS (
  SELECT item_id, COALESCE(MAX(opcao),0) AS max_opcao
  FROM item_opcoes
  WHERE item_id IN (SELECT item_id FROM novos)
  GROUP BY item_id
)
SELECT n.item_id, n.numero, n.nome, n.descricao, n.preco,
       COALESCE(b.max_opcao,0) + ROW_NUMBER() OVER (PARTITION BY n.item_id ORDER BY n.numero) AS opcao, n.parte
FROM novos n
LEFT JOIN base b USING (item_id);


END;

  CREATE OR REPLACE FUNCTION public.atualizavalorfracionado(p_user integer, p_pedido_id integer)
 RETURNS void
 LANGUAGE plpgsql
AS $function$

BEGIN
  -- 1. Atualiza numero
  UPDATE item_pedido_temp i
  SET numero = c.numero
  FROM cardapio c
  WHERE i.status = 'inicio'
    AND c.categoria = 'pizza'
    AND c.nome ILIKE '%'    i.nome    '%'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido_id
    AND i.numero = 0
    AND i.fracionada = 'true';

  -- 2. Atualiza numero_fracao
  UPDATE item_pedido_temp i
  SET numero_fracao = c.numero
  FROM cardapio c
  WHERE i.status = 'inicio'
    AND c.categoria = 'pizza'
    AND LOWER(c.nome) ILIKE '%'    LOWER(i.nome_fracao)    '%'
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido_id
    AND i.numero_fracao = 0
    AND i.fracionada = 'true';

  -- 3. Atualiza valor da primeira metade
  UPDATE item_pedido_temp i
  SET valor = c.preco_grande,
      valor_unitario = c.preco_grande,
      nome = '1/2 ('    i.numero::text    ') '    c.nome
  FROM cardapio c
  WHERE i.numero = c.numero
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido_id
    AND i.fracionada = 'true'
    AND i.categoria = 'pizza';

  -- 4. Atualiza valor_fracao
  UPDATE item_pedido_temp i
  SET valor_fracao = c.preco_grande,
      nome = i.nome    ' 1/2 ('    i.numero_fracao::text    ') '    c.nome
  FROM cardapio c
  WHERE i.numero_fracao = c.numero
    AND i.user_id = p_user
    AND i.pedido_id = p_pedido_id
    AND i.fracionada = 'true'
    AND i.categoria = 'pizza';

  -- 5. Mantém maior valor
  UPDATE item_pedido_temp i
  SET valor = CASE WHEN i.valor < i.valor_fracao THEN i.valor_fracao ELSE i.valor END * quantidade
  WHERE i.user_id = p_user
    AND i.pedido_id = p_pedido_id
    AND i.fracionada = 'true'
    AND i.categoria = 'pizza';
END;
$function$

  CREATE OR REPLACE FUNCTION public.match_documents(query_embedding vector, match_count integer DEFAULT NULL::integer, filter jsonb DEFAULT '{}'::jsonb)
 RETURNS TABLE(id bigint, content text, metadata jsonb, similarity double precision)
 LANGUAGE plpgsql
AS $function$
#variable_conflict use_column
BEGIN
  RETURN QUERY
  SELECT id, content, metadata,
         1 - (documents.embedding <=> query_embedding) AS similarity
    FROM documents
   WHERE metadata @> filter
   ORDER BY documents.embedding <=> query_embedding
   LIMIT match_count;
END; $function$

  CREATE OR REPLACE PROCEDURE public.reordenaritens(IN p_user_id bigint, IN p_pedido_id bigint)
 LANGUAGE plpgsql
AS $procedure$
BEGIN
  WITH ordenado AS (
    SELECT 
      item_id,
      user_id,
      pedido_id,
      ROW_NUMBER() OVER (
        PARTITION BY user_id, pedido_id
        ORDER BY item_id
      ) AS nova_ordem
    FROM item_pedido_temp
    WHERE user_id   = p_user_id
      AND pedido_id = p_pedido_id
      AND status    = 'concluido'   -- <<< filtra aqui!
  )
  UPDATE item_pedido_temp i
     SET ordem_item = o.nova_ordem
  FROM ordenado o
  WHERE i.item_id   = o.item_id
    AND i.user_id   = o.user_id
    AND i.pedido_id = o.pedido_id;
END;
