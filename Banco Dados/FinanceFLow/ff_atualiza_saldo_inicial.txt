CREATE OR REPLACE FUNCTION contab.ff_salvar_saldo_inicial(
    p_empresa_id bigint,
    p_conta_id   bigint,
    p_saldo      numeric(12,2)
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
    IF EXISTS (
        SELECT 1
        FROM contab.saldos_iniciais
        WHERE empresa_id = p_empresa_id
          AND conta_id   = p_conta_id
    ) THEN
        UPDATE contab.saldos_iniciais
           SET saldo = p_saldo
         WHERE empresa_id = p_empresa_id
           AND conta_id   = p_conta_id;
    ELSE
        INSERT INTO contab.saldos_iniciais (
            empresa_id,
            conta_id,
            saldo
        )
        VALUES (
            p_empresa_id,
            p_conta_id,
            p_saldo
        );
    END IF;
END;
$$;
