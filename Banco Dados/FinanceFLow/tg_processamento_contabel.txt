
--1   Transacoes 

 CREATE OR REPLACE FUNCTION contab.trg_transacoes_reprocesso()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN

  PERFORM contab.marcar_reprocessamento(
    NEW.empresa_id,
    NEW.data_movimento
  );

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_transacoes_reprocesso
AFTER INSERT ON public.transacoes
FOR EACH ROW
EXECUTE FUNCTION contab.trg_transacoes_reprocesso();


-- 2 CONTAS A RECEBER  
   CREATE OR REPLACE FUNCTION contab.trg_contas_receber_reprocesso()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
  v_data_base date;
BEGIN
  IF TG_OP = 'INSERT' THEN
    v_data_base := NEW.criado_em::date;

  ELSIF TG_OP = 'UPDATE' THEN
    IF OLD.data_recebimento IS DISTINCT FROM NEW.data_recebimento THEN
      v_data_base := COALESCE(NEW.data_recebimento::date, CURRENT_DATE);
    ELSE
      v_data_base := NULL;
    END IF;
  END IF;

  IF v_data_base IS NOT NULL THEN
    PERFORM contab.marcar_reprocessamento(NEW.empresa_id, v_data_base);
  END IF;

  RETURN NEW;
END;
$$;



--  3  Contas a pagar
  
 CREATE OR REPLACE FUNCTION contab.trg_contas_pagar_reprocesso()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
  v_data_base date;
BEGIN
  IF TG_OP = 'INSERT' THEN
    v_data_base := LEAST(
      COALESCE(NEW.vencimento::date, '9999-12-31'::date),
      COALESCE(NEW.data_pagamento::date, '9999-12-31'::date),
      COALESCE(NEW.criado_em::date, '9999-12-31'::date)
    );

  ELSIF TG_OP = 'UPDATE' THEN
    v_data_base := LEAST(
      COALESCE(NEW.vencimento::date, '9999-12-31'::date),
      COALESCE(NEW.data_pagamento::date, '9999-12-31'::date)
    );
  END IF;

  IF v_data_base <> '9999-12-31'::date THEN
    PERFORM contab.marcar_reprocessamento(
      NEW.empresa_id,
      v_data_base
    );
  END IF;

  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_contas_pagar_reprocesso ON public.contas_a_pagar;

CREATE TRIGGER trg_contas_pagar_reprocesso
AFTER INSERT OR UPDATE ON public.contas_a_pagar
FOR EACH ROW
EXECUTE FUNCTION contab.trg_contas_pagar_reprocesso();

--  4    compra no cartão 
CREATE OR REPLACE FUNCTION contab.trg_cartoes_compras_reprocesso()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN

  IF NEW.data_compra IS NOT NULL THEN
    PERFORM contab.marcar_reprocessamento(
      NEW.empresa_id,
      NEW.data_compra
    );
  END IF;

  RETURN NEW;
END;
$$;

--   compra no cartão 
CREATE TRIGGER trg_cartoes_compras_reprocesso
AFTER INSERT ON public.cartoes_compras
FOR EACH ROW
EXECUTE FUNCTION contab.trg_cartoes_compras_reprocesso();

-- 5 fatura no cartao 
CREATE OR REPLACE FUNCTION contab.trg_fatura_cartao_reprocesso()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN

  IF NEW.status = 'paga' AND NEW.data_pagamento IS NOT NULL THEN
    PERFORM contab.marcar_reprocessamento(
      NEW.empresa_id,
      NEW.data_pagamento
    );
  END IF;

  RETURN NEW;
END;
$$;

CREATE TRIGGER trg_fatura_cartao_reprocesso
AFTER INSERT OR UPDATE ON public.cartoes_faturas
FOR EACH ROW
EXECUTE FUNCTION contab.trg_fatura_cartao_reprocesso();
