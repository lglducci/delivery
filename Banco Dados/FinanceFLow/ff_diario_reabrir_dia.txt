 CREATE OR REPLACE FUNCTION contab.ff_diario_reabrir_periodo(
  p_empresa_id bigint,
  p_data_ini date,
  p_data_fim date
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
  DELETE FROM contab.diario
   WHERE empresa_id = p_empresa_id
     AND data_mov BETWEEN p_data_ini AND p_data_fim
     AND status = 'rascunho';
END;
$$;
