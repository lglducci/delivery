CREATE OR REPLACE FUNCTION contab.ff_dre_gerencial_analitico(
  p_empresa_id BIGINT,
  p_data_ini   DATE,
  p_data_fim   DATE
)
RETURNS TABLE (
  grupo TEXT,
  classificacao_gerencial TEXT,
  conta_codigo TEXT,
  conta_nome TEXT,
  valor NUMERIC(18,2)
)
LANGUAGE sql
AS $$
SELECT
  c.tipo AS grupo,
  c.classificacao_gerencial,
  c.codigo AS conta_codigo,
  c.nome AS conta_nome,
  CASE
    WHEN c.tipo = 'RECEITA' THEN (SUM(l.credito) - SUM(l.debito))
    WHEN c.tipo IN ('CUSTO', 'DESPESA') THEN (SUM(l.debito) - SUM(l.credito))
    ELSE 0
  END::numeric(18,2) AS valor
FROM contab.lancamentos l
JOIN contab.contas c
  ON c.id = l.conta_id
WHERE l.empresa_id = p_empresa_id
  AND l.data_mov BETWEEN p_data_ini AND p_data_fim
  AND c.analitica = true
  AND c.tipo IN ('RECEITA', 'CUSTO', 'DESPESA')
GROUP BY
  c.tipo,
  c.classificacao_gerencial,
  c.codigo,
  c.nome
HAVING SUM(l.debito) <> 0 OR SUM(l.credito) <> 0
ORDER BY c.tipo, c.codigo;
$$;
