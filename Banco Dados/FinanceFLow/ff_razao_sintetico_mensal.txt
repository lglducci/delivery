  CREATE OR REPLACE FUNCTION contab.ff_razao_sintetico_mensal (
    p_empresa_id BIGINT,
    p_data_ini   DATE,
    p_data_fim   DATE,
    p_filtro     TEXT DEFAULT NULL
)
RETURNS TABLE (
    conta_id       BIGINT,
    mes_ano        TEXT,
    conta_codigo   TEXT,
    conta_nome     TEXT,
    saldo_inicial  NUMERIC(14,2),
    valor          NUMERIC(14,2),
    saldo          NUMERIC(14,2)
)
LANGUAGE sql
STABLE
AS $$

WITH contas_base AS (
  SELECT
    c.id        AS conta_id,
    c.codigo    AS conta_codigo,
    c.nome      AS conta_nome,
    c.natureza
  FROM contab.contas c
  WHERE c.empresa_id = p_empresa_id
  AND (
      p_filtro IS NULL
      OR btrim(p_filtro) = ''
      OR c.codigo ILIKE '%' || p_filtro || '%'
      OR c.nome   ILIKE '%' || p_filtro || '%'
  )
),

saldo_base AS (
SELECT
  c.conta_id,
  c.conta_codigo,
  c.conta_nome,

  COALESCE(si.saldo,0)
  +
  COALESCE((
      SELECT     SUM(  CASE
  WHEN c.natureza='D' THEN l.debito - l.credito
  ELSE l.credito - l.debito
END  )
      FROM contab.lancamentos l
      WHERE l.empresa_id = p_empresa_id
      AND l.conta_id     = c.conta_id
      AND l.data_mov    < p_data_ini
  ),0) AS saldo_inicial

FROM contas_base c

LEFT JOIN contab.saldos_iniciais si
ON si.empresa_id = p_empresa_id
AND si.conta_id  = c.conta_id
),

movimento_periodo AS (

SELECT
  l.conta_id,

  SUM(  CASE
  WHEN c.natureza='D' THEN l.debito - l.credito
  ELSE l.credito - l.debito
END  ) AS valor

FROM contab.lancamentos l
JOIN contab.contas c
ON c.id = l.conta_id

WHERE l.empresa_id = p_empresa_id
AND l.data_mov BETWEEN p_data_ini AND p_data_fim

GROUP BY l.conta_id
)

SELECT
  s.conta_id,
  to_char(p_data_ini,'MM/YYYY') AS mes_ano,
  s.conta_codigo,
  s.conta_nome,
  s.saldo_inicial,
  COALESCE(m.valor,0)::numeric(14,2) AS valor,
  (s.saldo_inicial + COALESCE(m.valor,0))::numeric(14,2) AS saldo

FROM saldo_base s

LEFT JOIN movimento_periodo m
ON m.conta_id = s.conta_id

ORDER BY s.conta_codigo;

$$;
