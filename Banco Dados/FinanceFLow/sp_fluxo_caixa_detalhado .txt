   CREATE OR REPLACE FUNCTION contab.sp_fluxo_caixa_detalhado (
    p_empresa_id BIGINT,
    p_data_ini   DATE,
    p_data_fim   DATE
)
RETURNS TABLE (
    data_mov        DATE,
    conta_codigo    TEXT,
    conta_nome      TEXT,
    historico       TEXT,
    entrada         NUMERIC(14,2),
    saida           NUMERIC(14,2),
    saldo_mov       NUMERIC(14,2)
)
LANGUAGE sql
AS $$
WITH param AS (
    SELECT
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM contab.contas c
                WHERE c.empresa_id = p_empresa_id
                  AND c.analitica = TRUE
                  AND c.nivel = 4
                  AND c.codigo LIKE '1.1.1.%'
            )
            THEN 4
            ELSE 3
        END AS nivel_fluxo
),

contas_caixa AS (
    SELECT
        c.id,
        c.codigo,
        c.nome
    FROM contab.contas c
    CROSS JOIN param p
    WHERE c.empresa_id = p_empresa_id
      AND c.tipo = 'ATIVO'
      AND c.analitica = TRUE
      AND (
            (p.nivel_fluxo = 3 AND c.nivel = 3 AND c.codigo LIKE '1.1.%')
         OR (p.nivel_fluxo = 4 AND c.nivel = 4 AND c.codigo LIKE '1.1.1.%')
      )
),

saldo_base AS (
    SELECT COALESCE(SUM(si.saldo), 0) AS saldo
    FROM contab.saldos_iniciais si
    WHERE si.empresa_id = p_empresa_id
      AND si.conta_id IN (SELECT id FROM contas_caixa)
),

movimentos_anteriores AS (
    SELECT COALESCE(SUM(COALESCE(l.debito, 0) - COALESCE(l.credito, 0)), 0) AS saldo
    FROM contab.lancamentos l
    WHERE l.empresa_id = p_empresa_id
      AND l.conta_id IN (SELECT id FROM contas_caixa)
      AND l.data_mov::date < p_data_ini
),

saldo_inicial AS (
    SELECT
        COALESCE(sb.saldo, 0) + COALESCE(ma.saldo, 0) AS saldo
    FROM saldo_base sb
    CROSS JOIN movimentos_anteriores ma
),

movimentos_periodo AS (
    SELECT
        l.data_mov::date AS data_mov,
        c.codigo         AS conta_codigo,
        c.nome           AS conta_nome,
        l.historico      AS historico,
        COALESCE(l.debito, 0)  AS entrada,
        COALESCE(l.credito, 0) AS saida,
        COALESCE(l.debito, 0) - COALESCE(l.credito, 0) AS saldo_mov
    FROM contab.lancamentos l
    JOIN contas_caixa c
      ON c.id = l.conta_id
    WHERE l.empresa_id = p_empresa_id
      AND l.data_mov::date BETWEEN p_data_ini AND p_data_fim
)

SELECT
    p_data_ini            AS data_mov,
    NULL::TEXT            AS conta_codigo,
    'SALDO INICIAL'       AS conta_nome,
    'SALDO INICIAL'       AS historico,
    0::NUMERIC(14,2)      AS entrada,
    0::NUMERIC(14,2)      AS saida,
    s.saldo               AS saldo_mov
FROM saldo_inicial s

UNION ALL

SELECT
    m.data_mov,
    m.conta_codigo,
    m.conta_nome,
    m.historico,
    m.entrada,
    m.saida,
    m.saldo_mov
FROM movimentos_periodo m

ORDER BY
    data_mov,
    conta_codigo NULLS FIRST,
    historico;
$$;
