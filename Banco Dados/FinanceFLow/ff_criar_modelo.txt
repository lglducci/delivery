 create or replace function contab.ff_criar_modelo
(
 p_empresa_id int, 
 p_nome text, 
 p_token text, 
 p_conta_cred int,
 p_conta_deb int,
 p_tipo_automacao text,
 p_tipo_evento text,
 p_tipo_classificacao text default null  
)
returns int 
language plpgsql
as $$
declare
    v_modelo_id int;
    v_assinatura text;
begin

    -- 🔥 monta assinatura única
    v_assinatura := p_empresa_id || '|' || p_tipo_evento || '|' || p_tipo_classificacao || '|' || p_conta_deb || '|' || p_conta_cred;

    -- 🔍 verifica se já existe
    select id into v_modelo_id
    from contab.modelos
    where empresa_id = p_empresa_id
      and assinatura = v_assinatura
    limit 1;

    -- ✅ se já existe, ATUALIZA nome/token (resolve seu problema)
    if v_modelo_id is not null then

        update contab.modelos
        set nome = p_nome,
            codigo = p_token,
            tipo_automacao = p_tipo_automacao,
            tipo_evento = p_tipo_evento,
            classificacao = p_tipo_classificacao
        where id = v_modelo_id;

        return v_modelo_id;
    end if;

    -- 🚀 cria modelo novo
    insert into contab.modelos (
        empresa_id, 
        nome, 
        codigo, 
        tipo_automacao,
        tipo_evento, 
        classificacao,
        assinatura
    )
    values (
        p_empresa_id, 
        p_nome, 
        p_token, 
        p_tipo_automacao,
        p_tipo_evento, 
        p_tipo_classificacao,
        v_assinatura
    )
    returning id into v_modelo_id;

    -- 🔵 insere linhas
    insert into contab.modelos_linhas 
        (empresa_id, modelo_id, conta_id, dc, fonte_valor, valor_fixo)
    values 
        (p_empresa_id, v_modelo_id, p_conta_deb, 'D', 'Total', null);

    insert into contab.modelos_linhas 
        (empresa_id, modelo_id, conta_id, dc, fonte_valor, valor_fixo)
    values 
        (p_empresa_id, v_modelo_id, p_conta_cred, 'C', 'Total', null);

    return v_modelo_id;

end;
$$;
