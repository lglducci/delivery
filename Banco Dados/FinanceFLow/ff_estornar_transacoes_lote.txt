 CREATE OR REPLACE FUNCTION public.ff_estornar_transacoes_lote(
  p_empresa_id bigint,
  p_transacoes jsonb
)
RETURNS jsonb
LANGUAGE plpgsql
AS $$
DECLARE
  r jsonb;
  v_transacao_id bigint;
  v_qtd int := 0;
BEGIN
  FOR r IN
    SELECT value
    FROM jsonb_array_elements(p_transacoes)
  LOOP
    v_transacao_id := r::text::bigint;

    PERFORM public.ff_estornar_transacao(
      p_empresa_id,
      v_transacao_id
    );

    v_qtd := v_qtd + 1;
  END LOOP;

  RETURN jsonb_build_object(
    'ok', true,
    'mensagem', 'Transações estornadas com sucesso',
    'quantidade', v_qtd
  );
END;
$$;
