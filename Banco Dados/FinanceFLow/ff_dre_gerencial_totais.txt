CREATE OR REPLACE FUNCTION contab.ff_dre_gerencial_totais(
  p_empresa_id BIGINT,
  p_data_ini   DATE,
  p_data_fim   DATE
)
RETURNS TABLE (
  receita_4x NUMERIC(18,2),
  custo_variavel_5x NUMERIC(18,2),
  custo_fixo_5x NUMERIC(18,2),
  despesa_variavel_6x NUMERIC(18,2),
  despesa_fixa_6x NUMERIC(18,2),
  nao_operacional_6x NUMERIC(18,2),
  margem_contribuicao NUMERIC(18,2),
  margem_percentual NUMERIC(18,4),
  lucro_operacional NUMERIC(18,2),
  resultado_final NUMERIC(18,2)
)
LANGUAGE sql
AS $$
WITH base AS (
  SELECT
    c.codigo,
    c.tipo,
    c.classificacao_gerencial,
    CASE
      WHEN c.tipo = 'RECEITA' THEN (SUM(l.credito) - SUM(l.debito))
      WHEN c.tipo IN ('CUSTO', 'DESPESA') THEN (SUM(l.debito) - SUM(l.credito))
      ELSE 0
    END::numeric(18,2) AS valor
  FROM contab.lancamentos l
  JOIN contab.contas c
    ON c.id = l.conta_id
  WHERE l.empresa_id = p_empresa_id
    AND l.data_mov BETWEEN p_data_ini AND p_data_fim
    AND c.analitica = true
    AND c.tipo IN ('RECEITA', 'CUSTO', 'DESPESA')
  GROUP BY
    c.codigo,
    c.tipo,
    c.classificacao_gerencial
),
totais AS (
  SELECT
    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '4%'
         OR classificacao_gerencial = 'receita'
    ), 0)::numeric(18,2) AS receita_4x,

    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '5%'
        AND classificacao_gerencial = 'custo_variavel'
    ), 0)::numeric(18,2) AS custo_variavel_5x,

    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '5%'
        AND classificacao_gerencial = 'custo_fixo'
    ), 0)::numeric(18,2) AS custo_fixo_5x,

    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '6%'
        AND classificacao_gerencial = 'despesa_variavel'
    ), 0)::numeric(18,2) AS despesa_variavel_6x,

    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '6%'
        AND classificacao_gerencial = 'despesa_fixa'
    ), 0)::numeric(18,2) AS despesa_fixa_6x,

    COALESCE(SUM(valor) FILTER (
      WHERE codigo LIKE '6%'
        AND classificacao_gerencial = 'nao_operacional'
    ), 0)::numeric(18,2) AS nao_operacional_6x
  FROM base
)
SELECT
  receita_4x,
  custo_variavel_5x,
  custo_fixo_5x,
  despesa_variavel_6x,
  despesa_fixa_6x,
  nao_operacional_6x,

  (
    receita_4x
    - custo_variavel_5x
    - despesa_variavel_6x
  )::numeric(18,2) AS margem_contribuicao,

  CASE
    WHEN receita_4x <> 0 THEN
      (
        (receita_4x - custo_variavel_5x - despesa_variavel_6x)
        / receita_4x
      )::numeric(18,4)
    ELSE 0
  END AS margem_percentual,

  (
    receita_4x
    - custo_variavel_5x
    - despesa_variavel_6x
    - custo_fixo_5x
    - despesa_fixa_6x
  )::numeric(18,2) AS lucro_operacional,

  (
    receita_4x
    - custo_variavel_5x
    - despesa_variavel_6x
    - custo_fixo_5x
    - despesa_fixa_6x
    - nao_operacional_6x
  )::numeric(18,2) AS resultado_final
FROM totais;
$$;
