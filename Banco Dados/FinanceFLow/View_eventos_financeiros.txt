      

 drop view vw_eventos_financeiros;

 
    
   CREATE OR REPLACE VIEW vw_eventos_financeiros AS

SELECT
    id,
    empresa_id,
    descricao,
    valor_total AS valor,
    'saida' AS tipo,
    parcelas,
     parcelas as  parcela_total,
    NULL AS vencimento,
  'nao' vencido, 
    data_compra AS data_movimento,
    evento_codigo AS evento_codigo,
    criado_em AS data_criacao,
    tipo_evento,
    'compra_cartao' as origem, 
    classificacao,
    'cartao_compra' AS tipo_operacao,
   0 as  conta_id, 0  as categoria_id , 0 as  fornecedor_id , 0 origem_id,
   ''  as status
 FROM cartoes_compras

UNION ALL

SELECT
    id,
    empresa_id,
    descricao,
    valor,
    'saida' AS tipo,
     parcela_num as parcelas ,
      parcelas    as  parcela_total,
    vencimento, 
CASE
    WHEN vencimento IS NOT NULL AND vencimento < CURRENT_DATE THEN 'sim'
    ELSE 'nao'
END AS vencido,
    vencimento AS  data_movimento,
    evento_codigo AS evento_codigo,
    criado_em AS data_criacao,
    tipo_evento,
   'conta_pagar' as origem, 
    classificacao,
    'conta_pagar' AS tipo_operacao,
    0 as conta_id,  
  categoria_id ,   
  fornecedor_id  ,
  0 origem_id ,
    status
FROM contas_a_pagar

UNION ALL

SELECT
    id,
    empresa_id,
    descricao,
    valor,
    'entrada' AS tipo,
    parcela_num as  parcelas,
    parcelas    as  parcela_total,
    vencimento,
 CASE
    WHEN vencimento IS NOT NULL AND vencimento < CURRENT_DATE THEN 'sim'
    ELSE 'nao'
END AS vencido,
    vencimento AS data_movimento,
    evento_codigo AS  evento_codigo,
    criado_em AS data_criacao,
    tipo_evento,
    'conta_receber' as origem,  
    classificacao,
    'conta_receber' AS tipo_operacao ,
    0 as conta_id,  categoria_id ,    fornecedor_id  ,
     0 origem_id ,
     status
FROM contas_a_receber

UNION ALL

SELECT
    id,
    empresa_id,
    descricao,
    valor,
    'entrada' AS tipo,
    0 AS parcelas,
     0  as  parcela_total,
    NULL AS vencimento,
'nao' vencido, 
    data_movimento AS data_movimento,
    evento_codigo AS  evento_codigo,
    criado_em AS data_criacao,
    tipo_evento,
    
     CASE
    WHEN pagar_id IS NOT NULL THEN 'conta_pagar'
    WHEN receber_id IS NOT NULL THEN 'conta_receber'
    WHEN fatura_id IS NOT NULL THEN 'fatura_cartao'
    WHEN origem_id IS NOT NULL THEN 'estorno'
    ELSE 'transacao' 
    
END AS ORIGEM ,
 classificacao,
    'transacao' AS tipo_operacao ,
    conta_id,  categoria_id ,   0 as fornecedor_id ,
   origem_id,
   ''  as status
FROM transacoes
union all   
SELECT
    id,
    empresa_id,
    'Fatura cartão' AS descricao,
    valor_total AS valor,
    'saida' AS tipo,
    1 AS parcelas,
   1  as  parcela_total,
    vencimento,
 CASE
    WHEN vencimento IS NOT NULL AND vencimento < CURRENT_DATE THEN 'sim'
    ELSE 'nao'
END AS vencido,
    vencimento AS data_movimento,
    evento_codigo,
    criado_em AS data_criacao,
    'financeiro' AS tipo_evento, 
     'fatura_cartao' as origem, 
    'despesa' AS classificacao,
    'fatura_cartao' AS tipo_operacao,
    0 AS conta_id,
    0 AS categoria_id,
    0 AS fornecedor_id,
   0 origem_id,
    status
FROM cartoes_faturas
WHERE status IN ('fechada','paga');
