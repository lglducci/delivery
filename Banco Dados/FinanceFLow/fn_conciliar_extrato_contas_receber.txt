CREATE OR REPLACE FUNCTION public.fn_conciliar_extrato_contas_receber(
    prm_empresa_id bigint,
    prm_conta_financeira_id bigint,
    prm_id_maior_que bigint DEFAULT 0,
    prm_conciliacao_id bigint DEFAULT NULL
)
RETURNS void
LANGUAGE plpgsql
AS $$
DECLARE
    r record;
    x record;
BEGIN

    FOR r IN
        SELECT *
        FROM public.conciliacao_financeira
        WHERE empresa_id = prm_empresa_id
          AND conta_financeira_id = prm_conta_financeira_id
          AND tipo = 'entrada'
          AND receber_id IS NULL
          AND id > prm_id_maior_que
  AND (
      prm_conciliacao_id IS NULL
      OR id = prm_conciliacao_id
  )
        ORDER BY data_mov, id
    LOOP

        x := NULL;

        SELECT
            cr.id AS receber_id,

            (
              50
              +
              CASE 
                  WHEN abs(cr.valor - abs(r.valor)) <= 0.05 THEN 30
                  WHEN abs(cr.valor - abs(r.valor)) <= 5 THEN 20
                  WHEN abs(cr.valor - abs(r.valor)) <= 20 THEN 10
                  ELSE 0
              END
              +
              CASE 
                  WHEN abs(cr.vencimento - r.data_mov) <= 3 THEN 20
                  WHEN abs(cr.vencimento - r.data_mov) <= 7 THEN 10
                  WHEN abs(cr.vencimento - r.data_mov) <= 15 THEN 5
                  WHEN abs(cr.vencimento - r.data_mov) <= 45 THEN 2
                  ELSE 0
              END
            )::numeric(5,2) AS score,

            concat(
              'status=', cr.status,
              '; cliente=sim',
              '; valor_conta=', cr.valor,
              '; valor_extrato=', abs(r.valor),
              '; dif_valor=', abs(cr.valor - abs(r.valor)),
              '; vencimento=', cr.vencimento,
              '; data_mov=', r.data_mov,
              '; dif_dias=', abs(cr.vencimento - r.data_mov)
            ) AS criterio

        INTO x
        FROM public.contas_a_receber cr
        WHERE cr.empresa_id = r.empresa_id
          AND cr.fornecedor_id = r.fornecedor_id
          AND COALESCE(cr.status, 'aberto') <> 'recebido'
          AND abs(cr.valor - abs(r.valor)) <= 20
          AND abs(cr.vencimento - r.data_mov) <= 45
          AND NOT EXISTS (
              SELECT 1
              FROM public.conciliacao_financeira ja
              WHERE ja.empresa_id = r.empresa_id
                AND ja.receber_id = cr.id
          )
        ORDER BY
          score DESC,
          abs(cr.valor - abs(r.valor)) ASC,
          cr.vencimento ASC,
          cr.id ASC
        LIMIT 1;

        IF x.receber_id IS NOT NULL AND x.score >= 70 THEN
            UPDATE public.conciliacao_financeira
               SET receber_id = x.receber_id,
                   match_score = x.score,
                   match_criterio = x.criterio,
                   tipo_evento = 'receber'
             WHERE id = r.id;
        END IF;

    END LOOP;

END;
$$;
