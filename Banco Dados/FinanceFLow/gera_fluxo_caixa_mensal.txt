  CREATE OR REPLACE FUNCTION contab.gera_fluxo_caixa_mensal (
    p_empresa_id BIGINT,
    p_data_ini   DATE,
    p_data_fim   DATE
)
RETURNS TABLE (
    ano           INT,
    mes           INT,
    saldo_inicial NUMERIC(14,2),
    entrada       NUMERIC(14,2),
    saida         NUMERIC(14,2),
    saldo_final   NUMERIC(14,2)
)
LANGUAGE sql
AS $$
WITH param AS (
    SELECT
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM contab.contas c
                WHERE c.empresa_id = p_empresa_id
                  AND c.analitica = TRUE
                  AND c.nivel = 4
                  AND c.codigo LIKE '1.1.1.%'
            )
            THEN 4
            ELSE 3
        END AS nivel_fluxo
),

contas_caixa AS (
    SELECT
        c.id,
        c.codigo,
        c.nome
    FROM contab.contas c
    CROSS JOIN param p
    WHERE c.empresa_id = p_empresa_id
      AND c.tipo = 'ATIVO'
      AND c.analitica = TRUE
      AND (
            (p.nivel_fluxo = 3 AND c.nivel = 3 AND c.codigo LIKE '1.1.%')
         OR (p.nivel_fluxo = 4 AND c.nivel = 4 AND c.codigo LIKE '1.1.1.%')
      )
),

saldo_base AS (
    SELECT COALESCE(SUM(si.saldo), 0) AS saldo
    FROM contab.saldos_iniciais si
    WHERE si.empresa_id = p_empresa_id
      AND si.conta_id IN (SELECT id FROM contas_caixa)
),

movimentos AS (
    SELECT
        l.data_mov::date AS data_mov,
        COALESCE(l.debito, 0)  AS entrada,
        COALESCE(l.credito, 0) AS saida,
        COALESCE(l.debito, 0) - COALESCE(l.credito, 0) AS mov_liquido
    FROM contab.lancamentos l
    WHERE l.empresa_id = p_empresa_id
      AND l.conta_id IN (SELECT id FROM contas_caixa)
      AND l.data_mov::date <= p_data_fim
),

meses AS (
    SELECT
        gs::date AS mes_ref,
        GREATEST(gs::date, p_data_ini) AS periodo_ini,
        LEAST((gs + INTERVAL '1 month - 1 day')::date, p_data_fim) AS periodo_fim
    FROM generate_series(
        date_trunc('month', p_data_ini)::date,
        date_trunc('month', p_data_fim)::date,
        interval '1 month'
    ) gs
),

mensal AS (
    SELECT
        EXTRACT(YEAR FROM m.mes_ref)::int AS ano,
        EXTRACT(MONTH FROM m.mes_ref)::int AS mes,
        m.periodo_ini,
        m.periodo_fim,
        COALESCE((
            SELECT SUM(x.entrada)
            FROM movimentos x
            WHERE x.data_mov BETWEEN m.periodo_ini AND m.periodo_fim
        ), 0) AS entrada,
        COALESCE((
            SELECT SUM(x.saida)
            FROM movimentos x
            WHERE x.data_mov BETWEEN m.periodo_ini AND m.periodo_fim
        ), 0) AS saida
    FROM meses m
)

SELECT
    me.ano,
    me.mes,
    (
        sb.saldo
        +
        COALESCE((
            SELECT SUM(x.mov_liquido)
            FROM movimentos x
            WHERE x.data_mov < me.periodo_ini
        ), 0)
    ) AS saldo_inicial,
    me.entrada,
    me.saida,
    (
        sb.saldo
        +
        COALESCE((
            SELECT SUM(x.mov_liquido)
            FROM movimentos x
            WHERE x.data_mov <= me.periodo_fim
        ), 0)
    ) AS saldo_final
FROM mensal me
CROSS JOIN saldo_base sb
ORDER BY me.ano, me.mes;
$$; _crypto_aead_det_decrypt
