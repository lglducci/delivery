CREATE VIEW contab.vw_contas_validas_compra_cartao AS
SELECT *
FROM contab.contas
WHERE empresa_id = :empresa_id
  AND analitica = true
  AND (
       codigo LIKE '3.%'     -- despesas
    OR codigo LIKE '1.1.%'   -- ativo circulante (estoque, antecipadas)
    OR codigo LIKE '1.2.%'   -- imobilizado
  );
