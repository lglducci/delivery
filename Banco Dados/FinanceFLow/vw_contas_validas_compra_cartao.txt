CREATE OR REPLACE VIEW contab.vw_contas_validas_compra_cartao AS
SELECT *
FROM contab.contas
WHERE analitica = true
  AND (
       codigo LIKE '5.%'     -- custos
    OR codigo LIKE '6.%'     -- despesas
    OR codigo LIKE '1.1.%'   -- ativo circulante
    OR codigo LIKE '1.2.%'   -- imobilizado
  );
