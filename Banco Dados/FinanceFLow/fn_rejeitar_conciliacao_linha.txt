CREATE OR REPLACE FUNCTION public.fn_rejeitar_conciliacao_linha(
    p_empresa_id bigint,
    p_conciliacao_id bigint
)
RETURNS jsonb
LANGUAGE plpgsql
AS $$
BEGIN

    UPDATE public.conciliacao_financeira
    SET importar = false,
        status_conciliacao = 'rejeitado',
        mensagem_conciliacao = 'Rejeitado manualmente pelo usuário'
    WHERE empresa_id = p_empresa_id
      AND id = p_conciliacao_id
      AND COALESCE(status_conciliacao, 'pendente') <> 'executado'
      AND COALESCE(tipo_evento, '') IN ('pagar', 'receber', 'fatura_cartao');

    RETURN jsonb_build_object(
        'ok', true,
        'id', p_conciliacao_id
    );

END;
$$;
