CREATE OR REPLACE FUNCTION contab.sp_saldo_fluxo_realizado_ate_data (
    p_empresa_id BIGINT,
    p_data_corte DATE
)
RETURNS NUMERIC(14,2)
LANGUAGE sql
AS $$
WITH param AS (
    SELECT
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM contab.contas c
                WHERE c.empresa_id = p_empresa_id
                  AND c.analitica = TRUE
                  AND c.nivel = 4
                  AND c.codigo LIKE '1.1.1.%'
            )
            THEN 4
            ELSE 3
        END AS nivel_fluxo
),
contas_caixa AS (
    SELECT c.id
    FROM contab.contas c
    CROSS JOIN param p
    WHERE c.empresa_id = p_empresa_id
      AND c.tipo = 'ATIVO'
      AND c.analitica = TRUE
      AND (
            (p.nivel_fluxo = 3 AND c.nivel = 3 AND c.codigo LIKE '1.1.%')
         OR (p.nivel_fluxo = 4 AND c.nivel = 4 AND c.codigo LIKE '1.1.1.%')
      )
),
saldo_base AS (
    SELECT COALESCE(SUM(si.saldo), 0) AS saldo
    FROM contab.saldos_iniciais si
    WHERE si.empresa_id = p_empresa_id
      AND si.conta_id IN (SELECT id FROM contas_caixa)
),
movimentos AS (
    SELECT
        l.data_mov::date AS data_mov,
        COALESCE(l.debito, 0) - COALESCE(l.credito, 0) AS mov_liquido
    FROM contab.lancamentos l
    WHERE l.empresa_id = p_empresa_id
      AND l.conta_id IN (SELECT id FROM contas_caixa)
)
SELECT
    (
        sb.saldo
        + COALESCE((
            SELECT SUM(m.mov_liquido)
            FROM movimentos m
            WHERE m.data_mov <= p_data_corte
        ), 0)
    )::numeric(14,2)
FROM saldo_base sb;
$$;
