 CREATE OR REPLACE FUNCTION contab.fn_balancete(
    p_empresa_id BIGINT,
    p_data_ini   DATE,
    p_data_fim   DATE
)
RETURNS TABLE (
    conta_id      BIGINT,
    codigo        TEXT,
    conta_nome    TEXT,
    saldo_inicial NUMERIC(14,2),
    total_debito  NUMERIC(14,2),
    total_credito NUMERIC(14,2),
    saldo         NUMERIC(14,2)
)
LANGUAGE sql
AS $$
WITH saldo_base AS (
    SELECT
        si.conta_id,
        COALESCE(si.saldo, 0) AS saldo_implantado
    FROM contab.saldos_iniciais si
    WHERE si.empresa_id = p_empresa_id
),
mov_antes AS (
    SELECT
        l.conta_id,
        COALESCE(SUM(l.debito), 0)  AS debito_antes,
        COALESCE(SUM(l.credito), 0) AS credito_antes
    FROM contab.lancamentos l
    WHERE l.empresa_id = p_empresa_id
      AND l.data_mov < p_data_ini
    GROUP BY l.conta_id
),
mov_periodo AS (
    SELECT
        l.conta_id,
        COALESCE(SUM(l.debito), 0)  AS debito_periodo,
        COALESCE(SUM(l.credito), 0) AS credito_periodo
    FROM contab.lancamentos l
    WHERE l.empresa_id = p_empresa_id
      AND l.data_mov BETWEEN p_data_ini AND p_data_fim
    GROUP BY l.conta_id
)
SELECT
    c.id AS conta_id,
    c.codigo,
    c.nome AS conta_nome,

    CASE
        WHEN c.natureza = 'D' THEN
            COALESCE(sb.saldo_implantado, 0)
            + COALESCE(ma.debito_antes, 0)
            - COALESCE(ma.credito_antes, 0)
        ELSE
            COALESCE(sb.saldo_implantado, 0)
            + COALESCE(ma.credito_antes, 0)
            - COALESCE(ma.debito_antes, 0)
    END AS saldo_inicial,

    COALESCE(mp.debito_periodo, 0)  AS total_debito,
    COALESCE(mp.credito_periodo, 0) AS total_credito,

    CASE
        WHEN c.natureza = 'D' THEN
            (
                COALESCE(sb.saldo_implantado, 0)
                + COALESCE(ma.debito_antes, 0)
                - COALESCE(ma.credito_antes, 0)
            )
            + COALESCE(mp.debito_periodo, 0)
            - COALESCE(mp.credito_periodo, 0)
        ELSE
            (
                COALESCE(sb.saldo_implantado, 0)
                + COALESCE(ma.credito_antes, 0)
                - COALESCE(ma.debito_antes, 0)
            )
            + COALESCE(mp.credito_periodo, 0)
            - COALESCE(mp.debito_periodo, 0)
    END AS saldo

FROM contab.contas c
LEFT JOIN saldo_base sb ON sb.conta_id = c.id
LEFT JOIN mov_antes ma  ON ma.conta_id = c.id
LEFT JOIN mov_periodo mp ON mp.conta_id = c.id
WHERE c.empresa_id = p_empresa_id
ORDER BY c.codigo;
$$;
