CREATE OR REPLACE FUNCTION public.atualizar_valores_itens_temp(p_categoria text, p_user_id integer, p_pedido_id integer)
RETURNS void LANGUAGE plpgsql AS $$
BEGIN
  UPDATE public.item_pedido_temp
     SET tamanho = 'media'
   WHERE pedido_id = p_pedido_id AND user_id = p_user_id
     AND categoria = 'pizza'
     AND lower(tamanho) IN ('médio','m','medio','média','médias');

  UPDATE public.item_pedido_temp
     SET tamanho = 'grande'
   WHERE pedido_id = p_pedido_id AND user_id = p_user_id
     AND categoria = 'pizza'
     AND lower(tamanho) = 'g';

  UPDATE public.item_pedido_temp AS i
     SET valor = CASE WHEN i.tamanho='grande' THEN c.preco_grande
                      WHEN i.tamanho='media'  THEN c.preco_medio
                      ELSE c.preco_grande END * quantidade,
         nome = '('||c.numero::text||') '||c.nome,
         valor_unitario = CASE WHEN i.tamanho='grande' THEN c.preco_grande
                               WHEN i.tamanho='media'  THEN c.preco_medio
                               ELSE c.preco_grande END
    FROM public.cardapio c
   WHERE i.numero=c.numero AND i.user_id=p_user_id
     AND i.categoria=p_categoria AND i.fracionada=FALSE
     AND i.pedido_id=p_pedido_id;
END; $$;

CREATE OR REPLACE FUNCTION public.atualizavalorfracionado(p_user integer, p_pedido_id integer)
RETURNS void LANGUAGE plpgsql AS $$
BEGIN
  UPDATE public.item_pedido_temp i
     SET valor=c.preco_grande, valor_unitario=c.preco_grande,
         nome=' 1/2 ('||i.numero::text||') '||c.nome
    FROM public.cardapio c
   WHERE i.numero=c.numero AND i.user_id=p_user
     AND i.categoria='pizza' AND i.fracionada=TRUE
     AND i.pedido_id=p_pedido_id;

  UPDATE public.item_pedido_temp i
     SET valor_fracao=c.preco_grande,
         nome=i.nome||' 1/2 ('||i.numero_fracao::text||') '||c.nome
    FROM public.cardapio c
   WHERE i.numero_fracao=c.numero AND i.user_id=p_user
     AND i.categoria='pizza' AND i.fracionada=TRUE
     AND i.pedido_id=p_pedido_id;

  UPDATE public.item_pedido_temp i
     SET valor=(CASE WHEN i.valor<i.valor_fracao THEN i.valor_fracao ELSE i.valor END)*quantidade
   WHERE i.user_id=p_user AND i.categoria='pizza'
     AND i.fracionada=TRUE AND i.pedido_id=p_pedido_id;
END; $$;

CREATE OR REPLACE FUNCTION public.busca_cardapio(numeros text)
RETURNS TABLE(numero bigint, nome text, descricao text, preco_grande numeric, preco_media numeric, tipo text)
LANGUAGE plpgsql AS $$
BEGIN
  RETURN QUERY
  SELECT n::bigint,
         COALESCE(c.nome,'Produto não encontrado'),
         COALESCE(c.descricao,'não encontrado.'),
         COALESCE(c.preco_grande::numeric,0),
         COALESCE(c.preco_medio::numeric,0),
         c.tipo
    FROM unnest(string_to_array(numeros, ',')::int8[]) n
    LEFT JOIN public.cardapio c ON c.numero = n;
END; $$;

CREATE OR REPLACE FUNCTION public.busca_cardapio_por_descricao()
RETURNS TABLE(numero bigint, nome text, descricao text, preco_grande real, preco_medio real, tipo text, valido boolean)
LANGUAGE plpgsql AS $$
BEGIN
  RETURN QUERY
  SELECT c.numero, COALESCE(c.nome,'Produto não encontrado'),
         COALESCE(c.descricao,'não encontrado.'),
         COALESCE(c.preco_grande,0), COALESCE(c.preco_medio,0),
         c.tipo, (c.tipo='bebida' AND c.disponivel)
    FROM public.cardapio c
   WHERE c.tipo='bebida';
END; $$;

CREATE OR REPLACE FUNCTION public.busca_cardapio_por_descricao(descricoes text[])
RETURNS TABLE(numero bigint, descricao text, preco_medio real, preco_grande real, tipo text, categoria text)
LANGUAGE plpgsql AS $$
BEGIN
  RETURN QUERY
  SELECT c.numero, c.descricao, c.preco_medio, c.preco_grande, c.tipo, c.categoria
    FROM public.cardapio c
   WHERE c.tipo='bebida'
     AND EXISTS (SELECT 1 FROM unnest(descricoes) d WHERE lower(c.descricao) LIKE '%'||lower(d)||'%');
END; $$;

CREATE OR REPLACE FUNCTION public.calcula_pre_pedido(p_user_id integer, p_pedido_id integer)
RETURNS text LANGUAGE plpgsql AS $$
DECLARE
  v_total numeric := 0;
  v_pagto text; v_itens text; v_resumo text; v_qtd integer := 0;
BEGIN
  SELECT COALESCE(SUM(i.valor),0)+3,
         MAX(p.tipo_cobranca),
         STRING_AGG(
           E'\n        '||TRIM(BOTH ' ' FROM
             CASE WHEN COALESCE(i.quantidade,1)>1 THEN i.quantidade::text||'x ' ELSE '' END||
             COALESCE(i.nome,'')||
             CASE WHEN COALESCE(i.tamanho,'')<>'' THEN ' ('||i.tamanho||')' ELSE '' END||
             CASE WHEN COALESCE(i.categoria,'')<>'' THEN ' - '||i.categoria ELSE '' END||
             ' - R$ '||TO_CHAR(COALESCE(i.valor,0),'FM999G999G990D00')
           ), '' ORDER BY i.categoria,i.nome),
         COUNT(*) FILTER (WHERE i.pedido_id IS NOT NULL)
    INTO v_total, v_pagto, v_itens, v_qtd
    FROM public.pedido_temp p
    LEFT JOIN public.item_pedido_temp i ON i.user_id=p.user_id AND i.pedido_id=p.pedido_id
   WHERE p.user_id=p_user_id AND p.pedido_id=p_pedido_id
   GROUP BY p.user_id,p.pedido_id;

  IF NOT FOUND THEN RETURN '📦 Pedido não encontrado para o usuário informado.'; END IF;
  IF v_qtd=0 THEN RETURN '📦 Ainda não há itens no pedido.'; END IF;

  v_resumo := '📦 Pedido nº '||'Em construção'||E'\n'||
              '📋 Resumo: '||COALESCE(v_itens,'')||E'\n'||
              '💳 Forma de pagamento: '||COALESCE(v_pagto,'não informado')||E'\n'||
              '📦 Entrega: R$3,00'||E'\n'||
              '💰 Total: R$ '||TO_CHAR(v_total,'FM999G999G990D00')||E'\n'||
              '📦 Total de Itens: '||TO_CHAR(v_qtd,'0');
  RETURN v_resumo;
END; $$;

CREATE OR REPLACE FUNCTION public.cancela_pre_pedido(p_user_id integer, p_pedido_id integer)
RETURNS void LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  DELETE FROM public.item_pedido_temp WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
  DELETE FROM public.pedido_temp      WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
END; $$;

CREATE OR REPLACE FUNCTION public.finalizar_pedido(p_user_id integer, p_pedido_id integer)
RETURNS text LANGUAGE plpgsql AS $$
DECLARE
  v_pedido_id bigint; v_total numeric; v_pagto text; v_itens text; v_resumo text; v_comentario text;
BEGIN
  SELECT SUM(i.valor)+3, MAX(p.tipo_cobranca),
         STRING_AGG(
           E'\n       '||TRIM(BOTH ' ' FROM
             CASE WHEN i.quantidade>1 THEN i.quantidade::text||'x ' ELSE '' END||
             i.nome||
             CASE WHEN COALESCE(i.tamanho,'')<>'' THEN ' ('||i.tamanho||')' ELSE '' END||
             CASE WHEN COALESCE(i.categoria,'')<>'' THEN ' - '||i.categoria ELSE '' END||
             ' - R$ '||TO_CHAR(i.valor,'FM999G999G990D00')
           ), '' ORDER BY i.categoria,i.nome),
         COALESCE(NULLIF(BTRIM(c.comentario),''),'—')
    INTO v_total, v_pagto, v_itens, v_comentario
    FROM public.item_pedido_temp i
    LEFT JOIN public.pedido_temp p ON p.user_id=i.user_id AND p.pedido_id=i.pedido_id
    LEFT JOIN public.comentario_pedido_temp c ON c.user_id=i.user_id AND c.pedido_id=i.pedido_id
   WHERE i.user_id=p_user_id AND i.pedido_id=p_pedido_id
   GROUP BY c.comentario;

  IF v_total IS NULL THEN RAISE EXCEPTION 'Pedido temporário % do usuário % não encontrado ou sem itens.', p_pedido_id, p_user_id; END IF;

  INSERT INTO public.pedidos (user_id,status,valor,resumo,tipo_cobranca)
  SELECT user_id,'recebido',COALESCE(valor,v_total),NULL,tipo_cobranca
    FROM public.pedido_temp
   WHERE user_id=p_user_id AND pedido_id=p_pedido_id
  RETURNING pedido_id INTO v_pedido_id;

  v_resumo := '📦 Pedido nº '||v_pedido_id||E'\n'||
              '📋 Resumo: '||COALESCE(v_itens,'')||E'\n'||
              '💳 Forma de pagamento: '||COALESCE(v_pagto,'não informado')||E'\n'||
              '📦 Entrega: R$3,00'||E'\n'||
              '💰 Total: R$ '||TO_CHAR(v_total,'FM999G999G990D00')||E'\n'||
              '💬 Comentário: '||v_comentario;

  INSERT INTO public.item_pedido (
    numero,user_id,quantidade,tamanho,fracionada,tipo,categoria,numero_fracao,
    nome,volume,status,valor,valor_fracao,pedido_id,valor_unitario
  )
  SELECT numero,user_id,quantidade,tamanho,fracionada,tipo,categoria,numero_fracao,
         nome,volume,status,valor,valor_fracao,v_pedido_id,valor_unitario
    FROM public.item_pedido_temp
   WHERE user_id=p_user_id AND pedido_id=p_pedido_id;

  INSERT INTO public.comentario_pedido (user_id,pedido_id,comentario)
  SELECT user_id,v_pedido_id,comentario
    FROM public.comentario_pedido_temp
   WHERE user_id=p_user_id AND pedido_id=p_pedido_id;

  UPDATE public.pedidos SET resumo=v_resumo WHERE pedido_id=v_pedido_id;

  DELETE FROM public.comentario_pedido_temp WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
  DELETE FROM public.item_pedido_temp       WHERE user_id=p_user_id AND pedido_id=p_pedido_id;
  DELETE FROM public.pedido_temp            WHERE user_id=p_user_id AND pedido_id=p_pedido_id;

  RETURN v_resumo;
END; $$;
